title: "Toast"

---

## Instalación

```bash
# Instalar dependencias necesarias
npm install framer-motion lucide-react

# Instalar los componentes
npx @stofli/ui add toast
npx @stofli/ui add button
```

## Uso

```tsx
import { Toast, ToastProvider, useToast } from "@stofli/ui";
import { Button } from "@stofli/ui";

// Usando el componente directamente
export function ToastExample() {
  return (
    <Toast
      defaultOpen
      title="Notificación"
      description="Este es un toast de ejemplo"
      variant="success"
    />
  );
}

// Usando el hook useToast para mostrar toasts desde cualquier lugar
export function ToastWithProvider() {
  return (
    <ToastProvider>
      <ToastButton />
    </ToastProvider>
  );
}

function ToastButton() {
  const { open } = useToast();

  return (
    <Button
      variant="success"
      onClick={() =>
        open({
          title: "Guardado",
          description: "Tu información se ha guardado correctamente",
          variant: "success",
        })
      }
    >
      Mostrar Toast
    </Button>
  );
}
```

### Múltiples Toasts en cola

El componente ahora admite mostrar múltiples toasts simultáneamente, organizados según su posición.

  }
>
```tsx
import { ToastProvider, useToast } from "@/components/sections/toast";

export default function ToastMultipleDemo() {
  const { open } = useToast();
  
  return (
    <div className="flex flex-wrap gap-2">
      <Button
        variant="primary"
        onClick={() => {
          // Mostrar varios toasts en secuencia
          open({
            title: "Primer Toast",
            description: "Este es el primer toast",
            position: "bottom-right",
            variant: "primary"
          });
          
          setTimeout(() => {
            open({
              title: "Segundo Toast",
              description: "Este es el segundo toast",
              position: "bottom-right",
              variant: "success"
            });
          }, 500);
          
          setTimeout(() => {
            open({
              title: "Tercer Toast",
              description: "Este es el tercer toast",
              position: "bottom-right",
              variant: "warning"
            });
          }, 1000);
        }}
      >
        Mostrar varios toasts
      </Button>
      
      <Button
        variant="secondary"
        onClick={() => {
          // Mostrar toasts en diferentes posiciones
          open({
            title: "Superior izquierda",
            description: "Posición top-left",
            position: "top-left",
            variant: "primary"
          });
```

### Usando botones para mostrar toasts

Puedes usar el hook `useToast` dentro del `ToastProvider` para mostrar toasts desde cualquier parte de tu aplicación.

```tsx
import { ToastProvider, useToast } from "@stofli/ui";
import { Button } from "@stofli/ui";

export default function ToastWithButtons() {
  return (
    <ToastProvider>
      <ToastButton />
    </ToastProvider>
  );
}

function ToastButton() {
  const { open } = useToast();

  return (
    <div className="flex flex-wrap gap-2">
      <Button
        variant="success"
        onClick={() =>
          open({
            title: "Notificación",
            description: "Operación completada correctamente",
            variant: "success",
          })
        }
      >
        Mostrar notificación
      </Button>

      <Button
        variant="danger"
        onClick={() =>
          open({
            title: "Error",
            description: "Ha ocurrido un error inesperado",
            variant: "danger",
          })
        }
      >
        Mostrar error
      </Button>
    </div>
  );
}
```

## Props

| Prop              | Tipo                                                                                              | Default          | Descripción                                                                               |
| ----------------- | ------------------------------------------------------------------------------------------------- | ---------------- | ----------------------------------------------------------------------------------------- |
| `open`            | `boolean`                                                                                         | -                | Estado controlado del toast                                                               |
| `defaultOpen`     | `boolean`                                                                                         | `false`          | Estado inicial del toast si no es controlado                                              |
| `onOpenChange`    | `(open: boolean) => void`                                                                         | -                | Función llamada cuando cambia el estado                                                   |
| `title`           | `string`                                                                                          | -                | Título del toast                                                                          |
| `description`     | `string`                                                                                          | -                | Descripción o mensaje del toast                                                           |
| `variant`         | `"default" \| "primary" \| "secondary" \| "success" \| "warning" \| "danger"`                     | `"default"`      | Variante de estilo                                                                        |
| `position`        | `"top-left" \| "top-right" \| "bottom-left" \| "bottom-right" \| "top-center" \| "bottom-center"` | `"bottom-right"` | Posición en la pantalla                                                                   |
| `duration`        | `number`                                                                                          | `5000`           | Duración en ms antes de cerrarse automáticamente (use `Infinity` para que no desaparezca) |
| `showCloseButton` | `boolean`                                                                                         | `true`           | Muestra un botón para cerrar el toast                                                     |
| `className`       | `string`                                                                                          | -                | Clases adicionales para el contenedor                                                     |
| `children`        | `React.ReactNode`                                                                                 | -                | Contenido personalizado del toast                                                         |
| `icon`            | `React.ReactNode`                                                                                 | -                | Icono para mostrar junto al contenido                                                     |
